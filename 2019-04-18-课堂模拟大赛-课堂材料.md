### 汇编代码对照

格式：

```
====================
c++原语句
原语句对应的汇编代码
====================
```

```c++
==========================
int i=2;

movl	$2, -4(%rbp)

===========================
i++和++i

movl	-4(%rbp), %ecx
addl	$1, %ecx
movl	%ecx, -4(%rbp)

===========================
i=5+(++i);

movl	-4(%rbp), %ecx
addl	$1, %ecx
movl	%ecx, -4(%rbp)
addl	$5, %ecx
movl	%ecx, -4(%rbp)

===========================
i=5+(i++);

movl	-4(%rbp), %ecx
movl	%ecx, %edx
addl	$1, %edx
movl	%edx, -4(%rbp)
addl	$5, %ecx
movl	%ecx, -4(%rbp)
===========================
i=+(i++)+(++i)-i;

movl	-4(%rbp), %ecx
movl	%ecx, %edx
addl	$1, %edx
movl	%edx, -4(%rbp)
movl	-4(%rbp), %edx
addl	$1, %edx
movl	%edx, -4(%rbp)
addl	%edx, %ecx 
subl	-4(%rbp), %ecx
movl	%ecx, -4(%rbp)

===========================

```
